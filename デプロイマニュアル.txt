# 1. 初期化
git init

# 2. 【重要】除外設定（PowerShellで確実に認識させる形式）
Set-Content -Path .gitignore -Value "*.ai" -Encoding UTF8

# 3. ファイル追加（.ai が除外されていることを確認）
git add .
git commit -m "Initial commit (excluded heavy files)"

# 4. メインブランチ確定
git branch -M main

# 5. リモート登録（URLを差し替えて使用）
git remote add origin [REMOTE_URL]

# 6. 強制プッシュ（GitHub側の初期ファイルを上書き）
git push -f origin main
